#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;


bool sortGoal(const vector<float>& v1, const vector<float>& v2) {
    return v1[0] < v2[0];
}

int main() {
    vector<vector<float> > table = {{1, 12, 2.3232, 180}, {3, 21, -1.2333, 120}, {2, 3, -1.2323, 67.21}};
    
    for (int i = 0; i < table.size(); i++) {
        cout << "New Info\n";
        for (int j = 0; j < table[i].size(); j++) {
            cout << table[i][j] << ", ";
        }
        cout << "\n";
    }
    
    sort(table.begin(), table.end(), sortGoal);
    
    
    for (int i = 0; i < table.size(); i++) {
        cout << "New Info\n";
        for (int j = 0; j < table[i].size(); j++) {
            cout << table[i][j] << ", ";
        }
        cout << "\n";
    }
    
    
    return 0;
}
